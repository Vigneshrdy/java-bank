class Account {
    double balance = 0.0;
    String name;
    void deposit(double trans) {
        this.balance += trans;
    }
}

class current extends Account {
    savings saved;
    static double over = 0.05;
    double over_amt;
    double od_max = 10000.00;
    current(String n, double bal) {
        this.balance = bal;
        this.name = n;
    }
    
    void withdraw(double trans) {
        if(trans < balance) {
            balance -= trans;
        } else if(trans + over_amt <= od_max){
            over_amt = trans;
        }
        else {
            System.out.println("Transaction failed");
        }
    }
    void linkSavings(savings s) {
        saved = s;
    }
}
class savings extends Account {
    current SharedCurrent;
    double transactions;
    static double max_trn = 50000.00;
    boolean fd = false;
    double fd_amt;
    static double fd_rate = 0.0825;
    double time;
    double maturity_amt;
    
    savings (String n, double bal) {
        if(bal >= 1000.00) {
            this.balance = bal;
            this.name = n;
        } else {
            System.out.println("Minimum balance must be 1000.00 to create a savings account");
        }
    }
    
    void withdraw(double trans) {
        if((trans < balance - 1000.00) & (transactions + trans <= max_trn)) {
            balance -= trans;
            transactions += trans;
        } else {
            System.out.println("Transaction failed. Insufficient fund");
        }
    }
    
    void LinkCurrent (current c) {
        SharedCurrent = c;
    }
}
